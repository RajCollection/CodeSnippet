/* File Download using node.js*/

//install 'npm install download-file --save'

var download = require('download-file')

var url = "http://i.imgur.com/G9bDaPH.jpg"

var options = {
    directory: "./apis/files/",
    filename: "cat.jpg"
}

download(url, options, function(err){
    if (err) throw err
    console.log("meow")
})
